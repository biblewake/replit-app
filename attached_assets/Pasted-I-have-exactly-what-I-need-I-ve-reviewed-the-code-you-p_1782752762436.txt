### 1. Fix `useAlarmPermission.ts`

The problem with Replit's plan is that it would unconditionally prompt the user. We only want to check the live AlarmKit status *if the user has already seen the onboarding prompt*. We do this by checking a flag (`@bible_wake_has_prompted_alarm`).

Update your `useAlarmPermission.ts` to look like this:

```typescript
import { useEffect, useState, useRef } from "react";
import { AppState, AppStateStatus, PermissionsAndroid, Platform } from "react-native";
import * as Notifications from "expo-notifications";
import AsyncStorage from "@react-native-async-storage/async-storage";
import { syncAlarmKitAuthIfDenied } from "@/lib/alarmKitScheduler";

// We expose a helper so the onboarding screen can set this flag when the user taps "Allow"
export const ALARM_PROMPT_FLAG = "@bible_wake_has_prompted_alarm";

export function useAlarmPermission(): {
  hasPermission: boolean;
  hasExactAlarmPermission: boolean;
  alarmKitAuthorized: boolean;
} {
  const [hasPermission, setHasPermission] = useState(true);
  const [hasExactAlarmPermission, setHasExactAlarmPermission] = useState(true);
  const [alarmKitAuthorized, setAlarmKitAuthorized] = useState(true);
  
  // Use a ref to prevent overlapping state updates during rapid foregrounding
  const isChecking = useRef(false);

  const checkPermission = async () => {
    if (isChecking.current) return;
    isChecking.current = true;

    try {
      // 1. Check Standard Notifications
      try {
        const result = await Notifications.getPermissionsAsync();
        const status = (result as unknown as { granted?: boolean }).granted;
        setHasPermission(status !== false);
      } catch {
        setHasPermission(true);
      }

      // 2. Check Android Exact Alarms
      if (Platform.OS === "android" && Number(Platform.Version) >= 31) {
        try {
          const granted = await PermissionsAndroid.check(
            "android.permission.SCHEDULE_EXACT_ALARM" as Parameters<typeof PermissionsAndroid.check>[0]
          );
          setHasExactAlarmPermission(granted);
        } catch {
          setHasExactAlarmPermission(true);
        }
      }

      // 3. Check iOS 26 AlarmKit SAFELY
      if (Platform.OS === "ios") {
        try {
          // ONLY check live status if the user has already passed the onboarding prompt.
          // This prevents the OS dialog from ambushing brand-new users before they are ready.
          const hasPrompted = await AsyncStorage.getItem(ALARM_PROMPT_FLAG);
          
          if (hasPrompted === "true") {
             const denied = await syncAlarmKitAuthIfDenied();
             setAlarmKitAuthorized(!denied);
          } else {
             // If they haven't been prompted yet, we assume authorized=true to not block UI 
             // prematurely. The actual request happens at step 25.
             setAlarmKitAuthorized(true);
          }
        } catch {
           setAlarmKitAuthorized(true);
        }
      }
    } finally {
      isChecking.current = false;
    }
  };

  useEffect(() => {
    checkPermission();

    const sub = AppState.addEventListener("change", (state: AppStateStatus) => {
      if (state === "active") {
        checkPermission();
      }
    });

    return () => sub.remove();
  }, []);

  return { hasPermission, hasExactAlarmPermission, alarmKitAuthorized };
}

```
